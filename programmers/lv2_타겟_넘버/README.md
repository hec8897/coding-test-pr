function solution(numbers: number[], target: number): number {
  const dp = new Map<string, number>();

  function dfs(index: number, sum: number): number {
    if (index === numbers.length) {
      return sum === target ? 1 : 0;
    }

    const key = `${index}-${sum}`;
    if (dp.has(key)) {
      return dp.get(key)!;

    const result =
      dfs(index + 1, sum + numbers[index]) +
      dfs(index + 1, sum - numbers[index]);
    dp.set(key, result);
    return result;
  }

  return dfs(0, 0);
}
